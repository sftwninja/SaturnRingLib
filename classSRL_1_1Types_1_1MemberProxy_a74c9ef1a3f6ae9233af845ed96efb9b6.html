Construct a new member proxy.

Used to construct a proxy between instance member function (class bound function) and event

// Define sample class
class TinyClass {
uint32_t counter;
public:
// Define proxy
SRL::Types::MemberProxy<> Proxy = SRL::Types::MemberProxy(this, &TinyClass::Test);
// Attach to v-blank in constructor
TinyClass() {
SRL::Core::OnVblank += &this->Proxy;
}
// De-attach from v-blank in destructor
~TinyClass() {
SRL::Core::OnVblank -= &this->Proxy;
}
// Do our stuff inside this function
void Test() {
this->counter++;
}
}
// Program start
int main() {
SRL::Core::Initialize(HighColor::Colors::Black);
// Create object
TinyClass tiny();
// Game loop
while(1);
}
